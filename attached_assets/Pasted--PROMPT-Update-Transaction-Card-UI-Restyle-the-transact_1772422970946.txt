## CATEGORY LABEL RENAME MAP

Rename the raw category string to a fun display name before rendering the tag.
Apply this mapping in a helper function:

```javascript
function getCategoryDisplay(category, merchantName = '') {
  const name = merchantName.toLowerCase();
  const cat  = category.toLowerCase();

  if (name.includes('starbucks') || name.includes('tim hortons') || name.includes('coffee'))
    return 'DAILY SURRENDER';
  if (name.includes('whole foods'))
    return 'WHOLE PAYCHECK';
  if (name.includes('doordash') || name.includes('uber eats') || name.includes('skip'))
    return 'FLAVOR CRIMES';
  if (name.includes('uber') || name.includes('lyft'))
    return 'UBER ADDICTION';
  if (cat.includes('subscription') || cat.includes('streaming'))
    return 'FORGOTTEN CHILDREN';
  if (cat.includes('dining') || cat.includes('restaurant') || cat.includes('food'))
    return 'FLAVOR CRIMES';
  if (cat.includes('groceries') || cat.includes('grocery'))
    return 'WHOLE PAYCHECK';
  if (cat.includes('transport') || cat.includes('ride'))
    return 'UBER ADDICTION';
  if (cat.includes('alcohol') || cat.includes('bar') || cat.includes('liquor'))
    return 'LIQUID COURAGE FUND';
  if (cat.includes('shopping') || cat.includes('clothing'))
    return 'RETAIL THERAPY';

  return category.toUpperCase();
}
```
